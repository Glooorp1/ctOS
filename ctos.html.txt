        function navigateTo(screen) {
            document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
            
            const screenMap = {
                'main': 'main-screen',
                'scanner': 'scanner-screen',
                'cameras': 'cameras-screen',
                'profiler': 'profiler-screen'
            };
            
            document.getElementById(screenMap[screen]).classList.add('active');
            
            // Вибрация при переходе (если поддерживается)
            if (navigator.vibrate) {
                navigator.vibrate(10);
            }
        }

        // Вывод в терминал
        function terminalLog(elementId, message, type = '') {
            const terminal = document.getElementById(elementId);
            const timestamp = new Date().toLocaleTimeString();
            terminal.innerHTML += `\n<span class="${type}">[${timestamp}] ${message}</span>`;
            terminal.scrollTop = terminal.scrollHeight;
        }

        // === СКАНЕР СЕТИ ===
        function scanNetwork() {
            const term = document.getElementById('scanner-terminal');
            term.innerHTML = '<span class="info">Инициализация протокола сканирования...</span>';
            
            const devices = [
                { ip: '192.168.1.1', port: 80, type: 'Роутер', vuln: 'КРИТИЧЕСКАЯ' },
                { ip: '10.0.0.15', port: 554, type: 'IP-камера', vuln: 'ВЫСОКАЯ' },

                { ip: '172.20.0.5', port: 1883, type: 'IoT-концентратор', vuln: 'ВЫСОКАЯ' },
                { ip: '192.168.2.200', port: 502, type: 'ПЛК (промышленный)', vuln: 'КРИТИЧЕСКАЯ' }
            ];

            let delay = 0;
            devices.forEach((device, i) => {
                setTimeout(() => {
                    const vulnColor = device.vuln === 'КРИТИЧЕСКАЯ' ? 'error' : 'warn';
                    term.innerHTML += `
                        <div style="margin:8px 0; padding:8px; border:1px solid rgba(0,255,65,0.2);">
                            <span class="success">[+] Устройство #${i+1}</span><br>

                            Уязвимость: <span class="${vulnColor}">${device.vuln}</span>
                        </div>
                    `;
                    term.scrollTop = term.scrollHeight;
                    
                    // Вибрация при обнаружении критической уязвимости
                    if (device.vuln === 'КРИТИЧЕСКАЯ' && navigator.vibrate) {
                        navigator.vibrate([50, 100, 50]);

                }, delay);

                term.innerHTML += '<br><span class="success">✔ Сканирование завершено. 7 устройств обнаружено.</span>';

        // === ПРОФАЙЛЕР ===
        function searchProfile() {
            const username = document.getElementById('username-input').value.trim();
            const term = document.getElementById('profiler-terminal');

            if (!username) {
                term.innerHTML = '<span class="error">[!] Ошибка: введите имя пользователя.</span>';
                if (navigator.vibrate) navigator.vibrate(200);
                return;
            }

            term.innerHTML = `<span class="info">Поиск "${username}" в 347 социальных сетях...</span>`;

            const platforms = [
                `https://vk.com/${username}`,
                `https://t.me/${username}`,
                `https://github.com/${username}`,
                `https://instagram.com/${username}`,
                `https://twitter.com/${username}`,
                `https://reddit.com/user/${username}`,
                `https://habr.com/ru/users/${username}`,
                `https://youtube.com/@${username}`,
                `https://linkedin.com/in/${username}`,
                `https://facebook.com/${username}`
            ];

            // Рандомно выбираем 4-6 платформ для имитации поиска
            const found = platforms.sort(() => 0.5 - Math.random()).slice(0, Math.floor(Math.random() * 3) + 4);

            setTimeout(() => {
                term.innerHTML += '<br><br><span class="success">✔ НАЙДЕНЫ СОВПАДЕНИЯ:</span>';
                found.forEach((url, i) => {
                    setTimeout(() => {
                        term.innerHTML += `<br><span class="info">🔗 ${url}</span>`;
                        term.scrollTop = term.scrollHeight;
                    }, i * 300);
                });

                setTimeout(() => {
                    term.innerHTML += `<br><br><span class="warn">⚠ Найдено ${found.length} совпадений.</span>`;
                    term.innerHTML += '<br><span class="success">Цифровой след подтверждён.</span>';
                }, found.length * 300 + 100);
            }, 1500);
        }

            }, 3800);
        }
